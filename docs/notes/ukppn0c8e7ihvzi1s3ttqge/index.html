class MyContextManager:
    def __init__(self):
        self.used = False
    def __enter__(self):
        print('entering')
        self.used = True
        return self
    def __exit__(self, exc_type, exc_val, exc_tb):
        print('exiting')

with MyContextManager() as cm:
    print('in context')
    assert cm.used
